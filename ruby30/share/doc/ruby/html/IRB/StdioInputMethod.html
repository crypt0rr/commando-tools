def eof?
  rs, = IO.select([@stdin], [], [], 0.00001)
  if rs and rs[0]
    c = @stdin.getc
    result = c.nil? ? true : false
    @stdin.ungetc(c) unless c.nil?
    result
  else # buffer is empty
    false
  end
end
