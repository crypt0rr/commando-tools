def initialize(value)
  # convert to unsigned long
  value = [value].pack("L").unpack("L").first
  raise "#{value.to_s(16)} is not a recognized result" unless @@map.has_key? value
  @value = value
end

def ==(other)
  if other.is_a?(SSPIResult)
    @value == other.value
  elsif other.is_a?(Fixnum)
    @value == @@map[other]
  else
    false
  end
end
