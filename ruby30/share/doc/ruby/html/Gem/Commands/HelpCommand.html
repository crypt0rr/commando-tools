def execute
  arg = options[:args][0]

  _, help = SUBCOMMANDS.find do |command,|
    begins? command, arg
  end

  if help
    if Symbol === help
      send help
    else
      say help
    end
    return
  end

  if options[:help]
    show_help

  elsif arg
    show_command_help arg

  else
    say Gem::Command::HELP
  end
end
