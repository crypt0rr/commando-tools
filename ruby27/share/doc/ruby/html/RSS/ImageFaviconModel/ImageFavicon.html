def size=(new_value)
  if @do_validate and !new_value.nil?
    new_value = new_value.strip
    unless AVAILABLE_SIZES.include?(new_value)
      attr_name = "#{IMAGE_PREFIX}:size"
      raise NotAvailableValueError.new(full_name, new_value, attr_name)
    end
  end
  set_size(new_value)
end
