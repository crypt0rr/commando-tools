def cache_path
  @cache_path ||= begin
    git_scope = "#{base_name}-#{uri_hash}"

    Plugin.cache.join("bundler", "git", git_scope)
  end
end

def install_path
  @install_path ||= begin
    git_scope = "#{base_name}-#{shortref_for_path(revision)}"

    Plugin.root.join("bundler", "gems", git_scope)
  end
end
