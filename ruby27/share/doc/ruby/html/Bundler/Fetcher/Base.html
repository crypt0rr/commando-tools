def fetch_uri
  @fetch_uri ||= begin
    if remote_uri.host == "rubygems.org"
      uri = remote_uri.dup
      uri.host = "index.rubygems.org"
      uri
    else
      remote_uri
    end
  end
end

def log_specs(debug_msg)
  if Bundler.ui.debug?
    Bundler.ui.debug debug_msg
  else
    Bundler.ui.info ".", false
  end
end
